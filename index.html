Q7. Write a function that accepts parameter n and returns factorial of n.

        function fact(n)
{
    if(n<0)
    {
        console.log("error");
    }
    else if(n === 0)
    {
        console.log(`factorial of ${n} is 1`);
    }
    else
    {

        var factorial =1;
        for(let i=1; i<=n; i++)
        {
            if(n>1)
                {
                    factorial *= i;
                }
        }
        console.log(factorial);
    
    }
}
fact(8);


output: 120
